---
title: FormMessage
component: FormMessage
description: Useful for displaying messages under form fields.
---

```js live
<FormMessage message="Invalid input value. Only letters are allowed." icon={<Icon name="alert-triangle" color="#FFF" />} />
```

<PropsTable
  data={{
    message: {
      type: 'string',
      description: 'Required. The message to show.',
    },
    variant: {
      type: 'error | warning | info | success',
      default: 'error',
      description: 'Optional. Defines which color to apply to the message and icon background if present.',
    },
    icon: {
      type: 'ReactNode',
      description: 'Optional. The icon to display with the message.',
    },
  }}
/>

## Examples

### Colors

The color prop accepts any color as string.

```js live
<FormMessage message="Error message." />
```

```js live
<FormMessage message="Warning message." variant="warning" />
```

```js live
<FormMessage message="Information message." variant="info" />
```

```js live
<FormMessage message="Success message." variant="success" />
```

### Icons

The icon prop accepts a ReactNode and displays it inside a colored container.

```js live
<FormMessage 
  message="Error message." 
  variant="error" 
  icon={<Icon name="alert-triangle" color="#FFF" />}
/>
```

```js live
<FormMessage 
  message="Warning message." 
  variant="warning"
  icon={<Icon name="alert-triangle" color="#FFF" />}
/>
```

```js live
<FormMessage 
  message="Information message." 
  variant="info"
  icon={<Icon name="alert-circle" color="#FFF" />}
/>
```

```js live
<FormMessage 
  message="Success message." 
  variant="success"
  icon={<Icon name="checkmark-circle-2" color="#FFF" />}
/>
```
