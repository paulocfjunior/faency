## Examples

The following examples use the Dropdown component as a helper to display different use cases.

### MenuItemCheckbox

```js live removeFragment
function CheckboxExample() {
  const [option1, setOption1] = React.useState(true)
  const [option2, setOption2] = React.useState(false)
  const [option3, setOption3] = React.useState(true)

  return (
    <DropdownMenu
      button={<Button variant="primary">Open Menu</Button>}
      menu={
        <Menu>
          <MenuItemCheckbox label="Option 1" checked={option1} onChange={setOption1} />
          <MenuItemCheckbox label="Option 2" checked={option2} onChange={setOption2} />
          <MenuItemCheckbox label="Option 3" checked={option3} onChange={setOption3} />
        </Menu>
      }
    />
  )
}
```

### MenuItemRadioGroup and MenuItemRadio

```js live removeFragment
function CheckboxExample() {
  const [value, setValue] = React.useState('2')

  return (
    <DropdownMenu
      button={<Button variant="primary">Open Menu</Button>}
      menu={
        <Menu>
          <MenuItemRadioGroup value={value} onChange={setValue}>
            <MenuItemRadio value="1" label="Radio item one" />
            <MenuItemRadio value="2" label="Radio item two" />
            <MenuItemRadio value="3" label="Radio item three" />
          </MenuItemRadioGroup>
        </Menu>
      }
    />
  )
}
```
