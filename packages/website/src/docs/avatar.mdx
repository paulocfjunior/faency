## Examples

### Avatar using non-existing image with fallback to initials

```js live
<Avatar src="">KS</Avatar>
```

### Avatar using non-existing image with fallback to custom initials

```js live
<Avatar src="">
  <Text textColor="white" marginBottom={1}>
    K
  </Text>
  <Text textColor="white" marginTop={1}>
    S
  </Text>
</Avatar>
```

### Avatar using existing image on an alternative variant

```js live removeFragment
function AlternativeVariant() {
  const darkMode = useDarkMode()
  const isDarkMode = darkMode.value

  return (
    <Box backgroundColor="dark" padding={2}>
      <Avatar src={getValidImageUrl()} backgroundColor="purple" variant={isDarkMode ? 'light' : 'dark'}>
        KS
      </Avatar>
    </Box>
  )
}
```
